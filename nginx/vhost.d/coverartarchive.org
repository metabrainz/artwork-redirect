## Start of configuration add by letsencrypt container
location ^~ /.well-known/acme-challenge/ {
    auth_basic off;
    auth_request off;
    allow all;
    root /usr/share/nginx/html;
    try_files $uri =404;
    break;
}
## End of configuration add by letsencrypt container
proxy_ignore_headers "Set-Cookie";
proxy_hide_header "Set-Cookie";
proxy_ignore_headers X-Accel-Expires Expires Cache-Control;
add_header                    X-Cache-Status $upstream_cache_status always;

proxy_cache_methods GET HEAD;

proxy_cache caa_cache;

proxy_cache_key $scheme$proxy_host$request_uri;

proxy_cache_valid 307 5m;
proxy_cache_valid 200 404 1m;
proxy_cache_valid any 5s;

# Serve from cache if currently refreshing
proxy_cache_use_stale updating error timeout http_500 http_502 http_503 http_504;
proxy_cache_background_update on;
proxy_cache_lock on;
proxy_cache_min_uses 1;
proxy_cache_revalidate off;
proxy_cache_convert_head on;

# Redirect pesky radios
location ~* .(jpg|jpeg|png|gif|pdf|jfif|pjpeg|pjp) {
    if ($pesky_radio) {
        rewrite ^ /redirect.jpg break;
    }

    if ($pesky_radio = 0) {
        proxy_pass http://coverartarchive.org;
        break;
    }
}

location /release {
    if ($pesky_radio) {
        rewrite ^ /redirect.json break;
    }
    if ($pesky_radio = 0) {
        proxy_pass http://coverartarchive.org;
        break;
    }
}
